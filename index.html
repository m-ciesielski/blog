void plansza (char tab [16][16])
{
    int i, j, k;
    for(i=0; i<16; ++i)
    {
          printf ("\n");
          printf ("|");
        for (j=0; j<16; ++j)
        {
        printf ("%c|" , tab[i][j]);
        }

        printf ("\n");

        for (k=0; k<16; ++k)
        {
                 printf ("--");
        }

    }

}
